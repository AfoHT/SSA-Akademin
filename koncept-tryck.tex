\newcommand{\smallfig}[4][0.4]{

    \centering
    \includegraphics[width=#1\textwidth]{#2}
    \caption{#3}
    \label{#4}
    \vspace{-10pt}
  \end{figure}
}

\newcommand{\multifig}[4][0.4]{
  \begin{figure}[H]
    \centering

    \caption{#3}
    \label{#4}
    \vspace{-10pt}
  \end{figure}
}
